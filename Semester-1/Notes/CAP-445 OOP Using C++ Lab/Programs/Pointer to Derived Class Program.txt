//Pointer to Derived Class Program-------
#include <iostream>
using namespace std;

class BaseClass{
public:
    int baseVar=10;
    void display(){
        cout<<"I am Base Class and Variable of base class is "<<baseVar<<endl;
    }
};


class DerivedClass : public BaseClass{
public:
    int DerivedVar;
     void display(){
        cout<<"I am Derived Class and Variable of base class is "<<DerivedVar<<endl;
    }
};


int main(){

    BaseClass * bscp;

    DerivedClass dsc;

    bscp = &dsc; //pointer base class pointing to derived class

    bscp->display();
    bscp->baseVar=300;
    bscp->display();

    return 0;

}
