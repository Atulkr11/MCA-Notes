#include <iostream>
using namespace std;

class Addition {
private:
    int x;
    int y;
public:
    void Add(int x, int y)
    {
        this->x=x;
        this->y=y;
    }
    
    void disp(){
        cout<<"Addition : "<<x+y<<endl;
    }
};



int main()
{
    Addition ad;
    ad.Add(10,30);
    ad.disp();
    return 0;
}
