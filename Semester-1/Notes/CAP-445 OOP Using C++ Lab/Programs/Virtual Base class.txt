//Virtual Base class
#include <iostream>
using namespace std;

class A {
public:
    void f1()
    {
        cout << "Class A"<<endl;
    }
};

class B : virtual public A {
};

class C : virtual public A {
};

};

{
    D d;
    d.f1();
    return 0;
